// иҝӣе…Ҙ tryAcquireAsync
private <T> RFuture<Long> tryAcquireAsync(long waitTime, long leaseTime, TimeUnit unit, long threadId) {
    if (leaseTime != -1) {
        return tryLockInnerAsync(waitTime, leaseTime, unit, threadId, RedisCommands.EVAL_LONG);
    }
    //еҪ“leaseTime = -1 ж—¶ еҗҜеҠЁ watch dogжңәеҲ¶
    RFuture<Long> ttlRemainingFuture = tryLockInnerAsync(waitTime,
                                            commandExecutor.getConnectionManager().getCfg().getLockWatchdogTimeout(),
                                            TimeUnit.MILLISECONDS, threadId, RedisCommands.EVAL_LONG);
    //жү§иЎҢе®Ңluaи„ҡжң¬еҗҺзҡ„еӣһи°ғ
    ttlRemainingFuture.onComplete((ttlRemaining, e) -> {
        if (e != null) {
            return;
        }

        if (ttlRemaining == null) {
            // watch dog 
            scheduleExpirationRenewal(threadId);
        }
    });
    return ttlRemainingFuture;
}

private void scheduleExpirationRenewal(long threadId) {
    ExpirationEntry entry = new ExpirationEntry();
    //е°ҶзәҝзЁӢж”ҫе…Ҙзј“еӯҳдёӯ
    ExpirationEntry oldEntry = EXPIRATION_RENEWAL_MAP.putIfAbsent(getEntryName(), entry);
    //з¬¬дәҢж¬ЎиҺ·еҫ—й”ҒеҗҺ дёҚдјҡиҝӣиЎҢе»¶жңҹж“ҚдҪң
    if (oldEntry != null) {
        oldEntry.addThreadId(threadId);
    } else {
        entry.addThreadId(threadId);
        
        // з¬¬дёҖж¬ЎиҺ·еҫ—й”Ғ е»¶жңҹж“ҚдҪң
        renewExpiration();
    }
}

// иҝӣе…Ҙ renewExpiration()
private void renewExpiration() {
    ExpirationEntry ee = EXPIRATION_RENEWAL_MAP.get(getEntryName());
    //еҰӮжһңзј“еӯҳдёҚеӯҳеңЁпјҢйӮЈдёҚеҶҚй”Ғз»ӯжңҹ
    if (ee == null) {
        return;
    }
    
    Timeout task = commandExecutor.getConnectionManager().newTimeout(new TimerTask() {
        @Override
        public void run(Timeout timeout) throws Exception {
            ExpirationEntry ent = EXPIRATION_RENEWAL_MAP.get(getEntryName());
            if (ent == null) {
                return;
            }
            Long threadId = ent.getFirstThreadId();
            if (threadId == null) {
                return;
            }
            
            //жү§иЎҢlua иҝӣиЎҢз»ӯжңҹ
            RFuture<Boolean> future = renewExpirationAsync(threadId);
            future.onComplete((res, e) -> {
                if (e != null) {
                    log.error("Can't update lock " + getName() + " expiration", e);
                    return;
                }
                
                if (res) {
                    //е»¶жңҹжҲҗеҠҹпјҢз»§з»ӯеҫӘзҺҜж“ҚдҪң
                    renewExpiration();
                }
            });
        }
        //жҜҸйҡ”internalLockLeaseTime/3=10з§’жЈҖжҹҘдёҖж¬Ў
    }, internalLockLeaseTime / 3, TimeUnit.MILLISECONDS);
    
    ee.setTimeout(task);
}

// й”ҒйҮҠж”ҫ
public void unlock() {
    try {
        get(unlockAsync(Thread.currentThread().getId()));
    } catch (RedisException e) {
        if (e.getCause() instanceof IllegalMonitorStateException) {
            throw (IllegalMonitorStateException) e.getCause();
        } else {
            throw e;
        }
    }
}

// иҝӣе…Ҙ unlockAsync(Thread.currentThread().getId()) ж–№жі• е…ҘеҸӮжҳҜеҪ“еүҚзәҝзЁӢзҡ„id
public RFuture<Void> unlockAsync(long threadId) {
    RPromise<Void> result = new RedissonPromise<Void>();
    //жү§иЎҢluaи„ҡжң¬ еҲ йҷӨkey
    RFuture<Boolean> future = unlockInnerAsync(threadId);

    future.onComplete((opStatus, e) -> {
        // ж— и®әжү§иЎҢluaи„ҡжң¬жҳҜеҗҰжҲҗеҠҹ жү§иЎҢcancelExpirationRenewal(threadId) ж–№жі•жқҘеҲ йҷӨEXPIRATION_RENEWAL_MAPдёӯзҡ„зј“еӯҳ
        cancelExpirationRenewal(threadId);

        if (e != null) {
            result.tryFailure(e);
            return;
        }

        if (opStatus == null) {
            IllegalMonitorStateException cause = new IllegalMonitorStateException("attempt to unlock lock, not locked by current thread by node id: "
                    + id + " thread-id: " + threadId);
            result.tryFailure(cause);
            return;
        }

        result.trySuccess(null);
    });

    return result;
}

// жӯӨж–№жі•дјҡеҒңжӯў watch dog жңәеҲ¶
void cancelExpirationRenewal(Long threadId) {
    ExpirationEntry task = EXPIRATION_RENEWAL_MAP.get(getEntryName());
    if (task == null) {
        return;
    }
    
    if (threadId != null) {
        task.removeThreadId(threadId);
    }

    if (threadId == null || task.hasNoThreads()) {
        Timeout timeout = task.getTimeout();
        if (timeout != null) {
            timeout.cancel();
        }
        EXPIRATION_RENEWAL_MAP.remove(getEntryName());
    }
}
